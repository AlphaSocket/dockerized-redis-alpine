## Configurable envvars
~~~
CONFIG_PATHS_TEMPLATES_REDIS='/usr/local/templates/redis.conf'
CONFIG_PATHS_CONFIG_REDIS='/etc/redis/redis.conf'
CONFIG_REDIS_PERSISTENCE='yes'
CONFIG_REDIS_PASS='redis-default-pass'
~~~
